        checkThatFieldIsAnnotatedWith(BotMatchRecord.class, "playedMatch", ManyToOne.class);
        checkThatFieldIsAnnotatedWith(BotMatchRecord.class, "playedMatch", JoinColumn.class);

        // Cardinality "1" implies mandatory, so nullable = false
        checkThatFieldIsAnnotatedWith(BotMatchRecord.class, "againstBotProfile", ManyToOne.class);
        checkThatFieldIsAnnotatedWith(BotMatchRecord.class, "againstBotProfile", JoinColumn.class);
